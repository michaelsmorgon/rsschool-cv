function bubbleSort(arr) {
  let sorted;
    
  do {
    sorted = false;
    arr.forEach((val, index) => {
      if (val > arr[index + 1]) {
        let temp = val;
        arr[index] = arr[index + 1];
        arr[index + 1] = temp;
        sorted = true;
      }
    })
  } while (sorted);
}
